Database Setup

Run this SQL in your Supabase SQL Editor to create the required tables.

-- 1. Celebrities table CREATE TABLE IF NOT EXISTS celebrities ( id TEXT PRIMARY KEY, name TEXT NOT NULL DEFAULT '', photos JSONB NOT NULL DEFAULT '[]', created_at TIMESTAMPTZ DEFAULT NOW() ); -- 2. Ratings table CREATE TABLE IF NOT EXISTS ratings ( id UUID PRIMARY KEY DEFAULT GEN_RANDOM_UUID(), participant_id TEXT NOT NULL, participant_name TEXT NOT NULL DEFAULT '', image_index INT NOT NULL, celeb_id TEXT NOT NULL DEFAULT '', celeb_name TEXT NOT NULL DEFAULT '', photo_index INT NOT NULL DEFAULT 0, rating INT NOT NULL, comment TEXT NOT NULL DEFAULT '', created_at TIMESTAMPTZ DEFAULT NOW() ); -- 3. Study config table (single-row KV store) CREATE TABLE IF NOT EXISTS study_config ( key TEXT PRIMARY KEY, value JSONB NOT NULL DEFAULT 'null', updated_at TIMESTAMPTZ DEFAULT NOW() ); -- 4. Disable RLS for anon access (dev mode) ALTER TABLE celebrities ENABLE ROW LEVEL SECURITY; ALTER TABLE ratings ENABLE ROW LEVEL SECURITY; ALTER TABLE study_config ENABLE ROW LEVEL SECURITY; CREATE POLICY "public_all" ON celebrities FOR ALL TO anon USING (true) WITH CHECK (true); CREATE POLICY "public_all" ON ratings FOR ALL TO anon USING (true) WITH CHECK (true); CREATE POLICY "public_all" ON study_config FOR ALL TO anon USING (true) WITH CHECK (true);
